\makeglossaries

% = Acronym Definition
\newcommand{\lsad}[3]{ % Label, Acronym, Definition
  \newacronym{#1}{\texttt{#2}}{#3}
}

% = Glossary Entry
\newcommand{\lsge}[3]{ % Label, Acronym, Definition
  \newglossaryentry{#1}{
    name=#2,
    description={#3}
  }
}

\lsge{redis}{Redis}{The most popular in-memory, key-value data store.}
